# Hướng dẫn push code git
<!-- Khởi tạo -->
1. git init
<!-- Tạo branch dev -->
2. git checkout -b "dev"
<!-- Add code -->
3. git add . 
<!-- Commit code -->
4. git commit -m "noi dung"
<!-- Push code -->
5. git push origin dev
